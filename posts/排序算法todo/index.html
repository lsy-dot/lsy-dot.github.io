快速排序

//快速排序
public static Random random = new Random();
public int[] quickSort(int[] nums,int l,int r)
{
    if(l>=r)return nums;
    int pos=quickSortOne(nums,l,r);
    quickSort(nums,l,pos-1);
    quickSort(nums,pos+1,r);
    return nums;
}
public int quickSortOne(int[] nums,int l,int r)
{
    //加入随机
    int pos=random.nextInt(r-l+1)+l;
    swap(nums,l,pos);
    int pivot=nums[l];
    while(l<r)
    {
        while(l<r&&nums[r]>=pivot)r--;
        nums[l]=nums[r];
        while(l<r&&nums[l]<pivot)l++;
        nums[r]=nums[l];
    }
    nums[l]=pivot;
    return l;
}
public void swap(int[] nums,int p,int q)
{
    int temp=nums[p];
    nums[p]=nums[q];
    nums[q]=temp;
}

//heap sort
public void adjust(int[] nums,int len,int index)
{
    //  << 要加括号，不然运算符优先级不够
    int l=(index<<1)+1;
    int r=(index<<1)+2;
    int maxIndex=index;
    //这里记住比较的是nums[maxIndex] 而不是 nums[index]
    if(l<len&&nums[l]>nums[maxIndex])maxIndex=l;
    if(r<len&&nums[r]>nums[maxIndex])maxIndex=r;
    if(maxIndex!=index){
        swap(nums,index,maxIndex);
        adjust(nums,len,maxIndex);
    }
}
public int[] heapSort(int[] nums)
{
    //这里的长度是nums.length/2-1,调整非子结点
    for(int i=nums.length/2-1;i>=0;i--)
        adjust(nums,nums.length,i);
    for(int i=nums.length-1;i>=1;i--)
    {
        //调整子节点
        swap(nums,0,i);
        adjust(nums,i,0);
    }
    return nums;
}

归并排序

//归并排序
    public  void mergeSort(int[] nums,int l,int r){
        if(l>=r)return ;
        //int mid=(l+r)>>1;
        //这里必须加括号   话说>>的优先级也太低了吧
        int mid=l+((r-l)>>1);
        mergeSort(nums,l,mid);
        mergeSort(nums,mid+1,r);
        mergeSortInorder(nums,l,mid,r);
    }

    private void mergeSortInorder(int[] nums,int l,int m,int r) {
        int[] temps=new int[r-l+1];
        int k=0,s1=l,s2=m+1;
        while(s1<=m&&s2<=r)
        {
            if(nums[s1]<nums[s2])temps[k++]=nums[s1++];
            else temps[k++]=nums[s2++];
        }
        while(s1<=m)temps[k++]=nums[s1++];
        while(s2<=r)temps[k++]=nums[s2++];
        k=0;
        for(int i=l;i<=r;i++)nums[i]=temps[k++];
    }

 //冒泡排序
	public void bubbleSort(int[] nums)
    {
        int n=nums.length;
        for(int i=0;i<n-1;i++)
        {
            boolean flag=false;
            for(int j=n-1;j>i;j--)
            {
                if(nums[j]<nums[j-1]){
                    swap(nums,j,j-1);
                    flag=true;
                }
            }
            if(!flag)break;
        }
    }
